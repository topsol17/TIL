def test_list():
    scores = [10, 20, 30]
    assert scores[0] == 10
    assert scores[1] == 20
    assert scores[2] == 30
    assert scores[-1] == 30
    assert scores[-2] == 20
    assert scores[:2] == [10, 20]
    assert scores[1:] == [20, 30]
    assert scores[1:2] == [20]
    assert scores[:-1] == [10, 20]
    # append 실험
    scores.append(5)
    assert scores == [10, 20, 30, 5]
    # 바꾸기
    scores[0] = 0
    assert scores == [0, 20, 30, 5]
